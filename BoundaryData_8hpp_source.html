45 /*!
46  * \brief Creates both the Jacobian and inverse Jacobian between Cartesian and
47  * spherical coordinates, and the coordinates themselves
48  *
49  * \details The `cartesian_to_spherical_jacobian` is
50  * \f$dx^i/d\tilde{x}^{\tilde j}\f$,
51  * where the Cartesian components are in order \f$x^i = \{x, y, z\}\f$
52  * and the spherical coordinates are
53  * \f$\tilde{x}^{\tilde j} = \{r, \theta, \phi\}\f$.
54  * The Cartesian coordinates given are the standard unit sphere coordinates:
55  *
56  * \f{align*}{
57  * x &= \cos(\phi) \sin(\theta)\\
58  * y &= \sin(\phi) \sin(\theta)\\
59  * z &= \cos(\theta)
60  * \f}
61  *
62  * \note These Jacobians are adjusted to improve regularity near the pole, in
63  * particular the \f$\partial \phi / \partial x^i\f$ components have been scaled
64  * by \f$\sin \theta\f$ (omitting a \f$1/\sin(\theta)\f$) and the
65  * \f$\partial x^i/\partial \phi\f$ components have been scaled by
66  * \f$1/\sin(\theta)\f$ (omitting a \f$\sin(\theta)\f$). The reason is that in
67  * most careful calculations, these problematic sin factors can actually be
68  * omitted because they cancel. In cases where they are actually required, they
69  * must be put in by hand.
70  */
71 void cartesian_to_spherical_coordinates_and_jacobians(
72  gsl::not_null<tnsr::I<DataVector, 3>*> unit_cartesian_coords,
73  gsl::not_null<jacobian_tensor*> cartesian_to_spherical_jacobian,
74  gsl::not_null<inverse_jacobian_tensor*>
75  inverse_cartesian_to_spherical_jacobian,
76  const Scalar<DataVector>& cos_phi, const Scalar<DataVector>& cos_theta,
77  const Scalar<DataVector>& sin_phi, const Scalar<DataVector>& sin_theta,
78  double extraction_radius) noexcept;

80 /*!
81  * \brief Computes the spacetime metric and its first derivative in the
82  * intermediate radial null coordinates
83  *
84  * \details These components are obtained by the steps in
85  * Section II-A of \cite Barkett2019uae, which is based on the computation from
86  * Section 4.3 of \cite Bishop1998uk. The most direct comparison is to be made
87  * with equation (31) of \cite Barkett2019uae, which gives the null metric
88  * components explicitly. The time derivative is then (using notation from
89  * equation (31) of \cite Barkett2019uae):
90  *
91  * \f{align}{
92  * \partial_{\bar u} g_{\bar u \bar \lambda} =
93  * \partial_{\bar u} g_{\bar \lambda \bar \lambda} =
94  * \partial_{\bar u} g_{\bar \lambda \bar A} &= 0 \\
95  * \partial_{\bar u} g_{\bar u \bar u} &=
96  * \partial_{\breve t} g_{\breve t \breve t} \\
97  * \partial_{\bar u} g_{\bar u \bar A} &=
98  * \frac{\partial \breve x^{\breve i}}{\partial \bar x^{\bar A}}\\
99  * g_{\breve i \breve t}
100  * \partial_{\bar u} g_{\bar A \bar B}
101  * &= \frac{\partial \breve x^{\breve i}}{\partial \bar x^{\bar A}}
102  * \frac{\partial \breve x^{\breve j}}{\partial \bar x^{\bar B}}
103  * g_{\breve i \breve j}
104  * \f}
105  */
106 void null_metric_and_derivative(
107  gsl::not_null<tnsr::aa<DataVector, 3, Frame::RadialNull>*> du_null_metric,
108  gsl::not_null<tnsr::aa<DataVector, 3, Frame::RadialNull>*> null_metric,
109  const jacobian_tensor& cartesian_to_spherical_jacobian,
110  const tnsr::aa<DataVector, 3>& dt_spacetime_metric,
111  const tnsr::aa<DataVector, 3>& spacetime_metric) noexcept;

184  * determinant equation for `bondi_r`, from (35) of \cite Barkett2019uae :
